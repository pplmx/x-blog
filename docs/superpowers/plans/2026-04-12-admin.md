- [ ] **Step 1: 创建 Admin 配置**

```python
# backend/app/admin.py
from fastapi import FastAPI
from fastapi_admin import admins
from fastapi_admin.providers.login import LoginProvider
from starlette.requests import Request
from app.database import engine, Base
from app import models


async def admin_provider():
    return LoginProvider(
        admin_model=admins,
        login_logo_url="https://python.org/logo.png",
    )


def init_admin(app: FastAPI):
    from fastapi_admin import Admin

    admin = Admin(
        app,
        engine,
        admin_secret="x-blog-admin-secret",
        providers=[admin_provider()],
        title="X-Blog Admin",
        description="X-Blog 后台管理",
    )

    @admin.include_js("/static/admin.js")
    async def custom_js():
        return """
        <script>
            console.log('X-Blog Admin loaded');
        </script>
        """

    return admin
```

### Task 4: 集成 Admin 到 FastAPI

**Files:**

- Modify: `backend/app/main.py`

- [ ] **Step 1: 添加 Admin 路由**

```python
# backend/app/main.py
from contextlib import asynccontextmanager
from fastapi import FastAPI
from app.database import engine, Base
from app import models
from app.routers import posts, categories, tags
from app.admin import init_admin


@asynccontextmanager
async def lifespan(app: FastAPI):
    Base.metadata.create_all(bind=engine)
    yield


app = FastAPI(title="X-Blog API", version="0.1.0", lifespan=lifespan)

# 初始化 Admin
init_admin(app)

app.include_router(posts.router)
app.include_router(categories.router)
app.include_router(tags.router)


@app.get("/")
def read_root():
    return {"message": "X-Blog API"}
```

```python
# backend/app/init_admin.py
import asyncio
from app.database import engine, Base, SessionLocal
from app import models
from fastapi_admin.admins import Admin as AdminModel


async def create_admin():
    Base.metadata.create_all(bind=engine)

    db = SessionLocal()
    try:
        # 检查是否已有管理员
        existing = db.query(AdminModel).filter(
            AdminModel.username == "admin"
        ).first()

        if not existing:
            admin = AdminModel(
                username="admin",
                password="admin123",  # 生产环境请使用强密码
                is_superuser=True,
            )
            db.add(admin)
            db.commit()
            print("Admin user created: admin / admin123")
        else:
            print("Admin user already exists")
    finally:
        db.close()


if __name__ == "__main__":
    asyncio.run(create_admin())
```
